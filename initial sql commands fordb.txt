CREATE DATABASE KnowledgeForum;

USE KnowledgeForum;

-- User table
CREATE TABLE User (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    email VARCHAR(255) UNIQUE,
    role_on_site VARCHAR(50),
    password VARCHAR(255)
);

-- Post table
CREATE TABLE Post (
    post_id INT AUTO_INCREMENT PRIMARY KEY,
    post_text TEXT,
    likes INT DEFAULT 0,
    dislikes INT DEFAULT 0,
    title VARCHAR(255),
    state VARCHAR(50),
    user_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id) ON DELETE CASCADE
);

-- Tag table
CREATE TABLE Tag (
    tag_id INT AUTO_INCREMENT PRIMARY KEY,
    tag_name VARCHAR(255)
);

-- Post to Tag mapping table
CREATE TABLE Post_to_Tag (
    post_id INT,
    tag_id INT,
    PRIMARY KEY (post_id, tag_id),
    FOREIGN KEY (post_id) REFERENCES Post(post_id) ON DELETE CASCADE,
    FOREIGN KEY (tag_id) REFERENCES Tag(tag_id) ON DELETE CASCADE
);

-- Bookmark table
CREATE TABLE Bookmark (
    bookmark_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    post_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id) ON DELETE CASCADE,
    FOREIGN KEY (post_id) REFERENCES Post(post_id) ON DELETE CASCADE
);

-- Comments table
CREATE TABLE Comments (
    comment_id INT AUTO_INCREMENT PRIMARY KEY,
    comment_text TEXT,
    likes INT DEFAULT 0,
    dislikes INT DEFAULT 0,
    user_id INT,
    post_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id) ON DELETE CASCADE,
    FOREIGN KEY (post_id) REFERENCES Post(post_id) ON DELETE CASCADE
);

-- Reputation table
CREATE TABLE Reputation (
    user_id INT PRIMARY KEY,
    points INT DEFAULT 0,
    rank VARCHAR(50),
    FOREIGN KEY (user_id) REFERENCES User(user_id) ON DELETE CASCADE
);

-- Reports table
CREATE TABLE Reports (
    report_id INT AUTO_INCREMENT PRIMARY KEY,
    report_text TEXT,
    user_id INT,
    post_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id) ON DELETE CASCADE,
    FOREIGN KEY (post_id) REFERENCES Post(post_id) ON DELETE CASCADE
);

-- Group Message table
CREATE TABLE Group_Message (
    grp_id INT AUTO_INCREMENT PRIMARY KEY,
    grp_name VARCHAR(255),
    message TEXT,
    message_id INT,
    sender_user_id INT,
    FOREIGN KEY (sender_user_id) REFERENCES User(user_id) ON DELETE CASCADE
);

-- Personal Message table
CREATE TABLE Personal_Message (
    message_id INT AUTO_INCREMENT PRIMARY KEY,
    sent_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    message TEXT,
    sender_user_id INT,
    receiver_user_id INT,
    FOREIGN KEY (sender_user_id) REFERENCES User(user_id) ON DELETE CASCADE,
    FOREIGN KEY (receiver_user_id) REFERENCES User(user_id) ON DELETE CASCADE
);
